building_senate_headquarters = {
    category = government

    base_buildtime = 360
    base_cap_amount = 1

    building_sets = {
        government_buildings
    }

    resources = {
        category = government_buildings
        cost = {
            minerals = 300
            alloys = 100
        }
        upkeep = {
            energy = 2
        }
    }

    potential = {
        owner = {
            has_authority = auth_democratic
            NOT = { has_ethic = ethic_gestalt_consciousness }
            NOT = { has_country_flag = built_senate_headquarters }
        }
        NOT = { has_modifier = slave_colony }
    }

    on_built = {
        owner = {
            set_country_flag = built_senate_headquarters
        }
    }

    on_destroy = {
        owner = {
            remove_country_flag = built_senate_headquarters
        }
    }

    triggered_planet_modifier = {
        potential = {
            exists = owner
        }
        modifier = {
            job_senator_add = 3
            planet_stability_add = 10
        }
    }

    triggered_desc = {
        trigger = {
            exists = owner
        }
        text = job_senator_effect_desc
    }
}
